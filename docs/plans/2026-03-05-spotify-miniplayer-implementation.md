**Step 3: Create `src-tauri/src/spotify_auth.rs`**

Implement:
- `generate_pkce_pair()` → returns `(code_verifier, code_challenge)`
- `start_auth_flow(client_id)` → opens browser to Spotify auth URL, starts local HTTP server on port 8888 to receive callback, returns authorization code
- `exchange_code_for_token(client_id, code, code_verifier)` → POST to `https://accounts.spotify.com/api/token`, returns access_token + refresh_token + expires_in
- `refresh_access_token(client_id, refresh_token)` → POST to token endpoint with grant_type=refresh_token

Scopes needed: `user-read-playback-state user-modify-playback-state user-read-currently-playing`

Token storage: Save to a JSON file at `~/.spotify-miniplayer/tokens.json` using `dirs` crate (add `dirs = "5"` to Cargo.toml).

**Step 4: Wire auth into Tauri as a command**

In `lib.rs`:

## Task 5: Connect Frontend to Backend

**Files:**
- Modify: `src/main.js`

**Step 1: Replace placeholder JS with real Tauri integration**

Implement in `src/main.js`:

```javascript
const { invoke } = window.__TAURI__.core;
const { getCurrentWebviewWindow } = window.__TAURI__.webviewWindow;

const appWindow = getCurrentWebviewWindow();

let isPlaying = false;
let pollInterval = null;
let progressInterval = null;
let currentProgress = 0;
let currentDuration = 0;
let fadeTimeout = null;

// Poll playback every 2 seconds
async function pollPlayback() {
  try {
    const state = await invoke('get_playback');

    if (state && state.is_playing) {
      updateUI(state);
      showWidget();
      clearFadeTimeout();
    } else if (state && !state.is_playing) {
      updateUI(state);
      startFadeTimeout();
    } else {
      startFadeTimeout();
    }
  } catch (e) {
    console.error('Poll failed:', e);
  }
}

function updateUI(state) {
  document.getElementById('track-name').textContent = state.track_name;
  document.getElementById('artist-name').textContent = state.artist_name;
  document.getElementById('album-art').src = state.album_art_url;
  currentProgress = state.progress_ms;
  currentDuration = state.duration_ms;
  updateProgress();
  updatePlayIcon(state.is_playing);
  isPlaying = state.is_playing;
}

// Local progress interpolation (smooth bar between polls)
function startProgressInterpolation() { ... }

// Show/hide widget with animation
function showWidget() {
  document.getElementById('widget').classList.add('visible');
  appWindow.show();
}

function hideWidget() {
  document.getElementById('widget').classList.remove('visible');
  setTimeout(() => appWindow.hide(), 300); // wait for fade animation
}

// Fade timeout (3s after pause)
function startFadeTimeout() {
  if (fadeTimeout) return;
  fadeTimeout = setTimeout(hideWidget, 3000);
}

// Button handlers
document.getElementById('btn-play').addEventListener('click', () => invoke('play_pause'));
document.getElementById('btn-next').addEventListener('click', () => invoke('next_track'));
document.getElementById('btn-prev').addEventListener('click', () => invoke('previous_track'));

// Progress bar click to seek
document.getElementById('progress-container').addEventListener('click', (e) => {
  const bar = e.currentTarget.querySelector('.progress-bar');
  const rect = bar.getBoundingClientRect();
  const pct = (e.clientX - rect.left) / rect.width;
  const posMs = Math.floor(pct * currentDuration);
  invoke('seek_to', { positionMs: posMs });
});

// Start polling
pollInterval = setInterval(pollPlayback, 2000);
pollPlayback(); // immediate first poll
startProgressInterpolation();
```
